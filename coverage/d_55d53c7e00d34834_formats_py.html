73 """Transform a command or function into a string. 

74

76 cmd: The command or function to transform.

90def as_shell_command(cmd: list[str]) -> str: 

91 """Rebuild a command line from system arguments. 

92 

93 Arguments: 

94 cmd: The command as a list of strings. 

95 

96 Returns: 

97 A printable and shell-runnable command. 

98 """ 

99 parts = [] 

100 for part in cmd: 

101 if not part: 

102 parts.append('""') 

103 continue 

104 has_spaces = " " in part 

105 has_double_quotes = '"' in part 

106 has_single_quotes = "'" in part 

107 if has_double_quotes and not has_single_quotes: 

108 # double quotes, no single quotes 

109 # -> wrap in single quotes 

110 part = f"'{part}'" # noqa: PLW2901 

111 elif has_single_quotes and has_double_quotes: 

112 # double and single quotes 

113 # -> escape double quotes, wrap in double quotes 

114 part = part.replace('"', r"\"") # noqa: PLW2901 

115 part = f'"{part}"' # noqa: PLW2901 

116 elif has_single_quotes or has_spaces: 

117 # spaces or single quotes 

118 # -> wrap in double quotes 

119 part = f'"{part}"' # noqa: PLW2901 

120 parts.append(part) 

121 return " ".join(parts)

124def as_python_statement(func: Callable | LazyCallable, args: Sequence | None = None, kwargs: dict | None = None) -> str: 

125 """Transform a callable and its arguments into a Python statement string. 

126 

127 Arguments: 

128 func: The callable to transform. 

129 args: Positional arguments passed to the function. 

130 kwargs: Keyword arguments passed to the function. 

131 

132 Returns: 

133 A Python statement. 

134 """ 

135 if isinstance(func, LazyCallable): 

136 callable_name = func.name or _get_callable_name(func.call) 

137 args = args or func.args 

138 kwargs = kwargs or func.kwargs 

139 else: 

140 callable_name = _get_callable_name(func) 

141 args_str = [repr(arg) for arg in args] if args else [] 

142 kwargs_str = [f"{k}={v!r}" for k, v in kwargs.items()] if kwargs else [] 

143 arguments = ", ".join(args_str + kwargs_str) 

144 return f"{callable_name}({arguments})"

147def _get_callable_name(callee: Callable) -> str: 

148 callable_name = getattr(callee, "__name__", None) 

149 if callable_name: 

150 return callable_name 

151 

152 # Climb back up the frames to search the callable in the locals 

153 callable_name = None 

154 caller_frame: FrameType = inspect.currentframe() # type: ignore[assignment] 

155 while callable_name is None and caller_frame.f_back: 

156 caller_frame = caller_frame.f_back 

157 callable_name = _find_callable_name_in_frame_locals(caller_frame, callee) 

158 

159 return callable_name or DEFAULT_CALLABLE_NAME

162def _find_callable_name_in_frame_locals(caller_frame: FrameType, callee: Callable) -> str | None: 

163 call_vars = caller_frame.f_locals.items() 

164 try: 

165 # ignore @py_assert variables from pytest 

166 return next(var_name for var_name, var_val in call_vars if var_val is callee and not var_name.startswith("@")) 

167 except StopIteration: 

168 return None
